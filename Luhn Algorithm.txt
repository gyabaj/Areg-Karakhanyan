#include <iostream>
using namespace std;
enum whichProblem {SWAPED, ABCENT};

bool abcent(int* arr, int sum, int i = 15)
{
    if (arr[i] == 100 || arr[i - 1] == 100)
    {
        if (arr[i] == 100)
            for (int j = 0; j < 10; ++j)
            {
                arr[i] = j;
                if (abcent(arr, sum, i))
                    return true;
            }
        if (arr[i - 1] == 100)
            for (int j = 0; j < 10; ++j)
            {
                arr[i - 1] = j;
                if (abcent(arr, sum, i))
                    return true;
            }
    }
    else
    {
        sum += arr[i];
        (arr[i - 1] * 2 > 9) ? sum += arr[i - 1] * 2 - 9 : sum += arr[i - 1] * 2;
        if (!(i - 1))
            return !(sum % 10);
        else 
            return (abcent(arr, sum, i - 2));
    }
    return false;
}
bool swapped(int* arr, int sum, bool dontSwap = false, int i = 15)
{
    int sum1 = sum;
    if (i % 2)
        sum1 += arr[i];
    else
        (arr[i] * 2 > 9) ? sum1 += arr[i] * 2 - 9 : sum1 += arr[i] * 2;
    if (!i)
        return (!(sum1 % 10));
    if (swapped(arr, sum1, dontSwap, i - 1))
        return true;
    else if (dontSwap)
        return false;
    else
    {
        swap(arr[i], arr[i - 1]);
        if (i % 2)
            sum += arr[i];
        else
            (arr[i] * 2 > 9) ? sum += arr[i] * 2 - 9 : sum += arr[i] * 2;
        if (swapped(arr, sum, true, i - 1))
            return true;
        else
        {
            swap(arr[i], arr[i - 1]);
            return false;
        }
    }
    return false;
}
int main()
{
    int n{};
    cin >> n;
    int arr[16]{}; 
    while (n--)
    {
        whichProblem typeOfProb{};
        for (int i = 0; i < 16; ++i)
        {
            char input{};
            cin >> input;
            if (input == '?')
            {
                arr[i] = 100;
                typeOfProb = ABCENT;
            }
            else if ('0' <= input && input <= '9')
                arr[i] = input - '0';
            else
                cout << "Wrong input";
        }
        if (typeOfProb == ABCENT)
            abcent(arr, 0);
        else
            swapped(arr,0);

        for (auto el : arr)
            cout << el;
        cout << ' ';
    }
}
